    def create_emotion_report(clusters_data, output_dir): """ Create reports and visualizations for emotion distributions in clusters

    # List of emotions and colors for consistency emotions = ['angry', 'sad', 'surprised', 'happy'] colors = ['red', 'blue', 'orange', 'green']

    for cluster in sorted(clusters_data.keys()): f.write(f"Cluster {cluster}: ") distributions = [] for emotion in emotions: percentage = clusters_data[cluster].get(emotion, 0) distributions.append(f"{emotion} {percentage:.1f}%") f.write(" | ".join(distributions) + "\n")

    # Create bar chart plt.figure(figsize=(12, 6)) x = np.arange(len(clusters_data)) width = 0.2
